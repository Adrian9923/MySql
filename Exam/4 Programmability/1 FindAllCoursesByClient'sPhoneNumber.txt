TASK: Create a user defined function with the name udf_courses_by_client (phone_num VARCHAR (20)) that receives a client's phone number and returns the number of courses that clients have in the database.




DELIMITER &&
CREATE FUNCTION `udf_courses_by_client`(`phone_num` VARCHAR(20)) RETURNS INT DETERMINISTIC
BEGIN
	DECLARE `get_phone_number_id` INT;
    
    SET `get_phone_number_id` = (
			SELECT `id` FROM `clients` WHERE `phone_number` = `phone_num`
    );
    
    RETURN (
		SELECT COUNT(`client_id`)
        FROM `courses`
        WHERE `client_id` = `get_phone_number_id`
    
    );
END; &&

SELECT udf_courses_by_client ('(831) 1391236') as 'count';
